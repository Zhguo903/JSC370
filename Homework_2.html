def get_country_only_list() -> set:
    """
    Get official World Bank 'country-only' names (exclude region/aggregate entries).
    """
    url = "https://api.worldbank.org/v2/country"
    params = {"format": "json", "per_page": 400}
    r = requests.get(url, params=params)
    r.raise_for_status()
    data = r.json()[1]
    meta = pd.DataFrame(data)

    meta["region_name"] = meta["region"].apply(
        lambda x: x.get("value") if isinstance(x, dict) else None
    )
    meta = meta[meta["region_name"] != "Aggregates"]
    return set(meta["name"].astype(str))

COUNTRY_ONLY = get_country_only_list()

def fetch_wb_indicator(indicator_code: str, value_name: str, years=YEARS) -> pd.DataFrame:
    url = f"https://api.worldbank.org/v2/country/all/indicator/{indicator_code}"
    params = {"format": "json", "per_page": 20000}
    r = requests.get(url, params=params)
    r.raise_for_status()
    payload = r.json()
    data = payload[1]

    df = pd.DataFrame(data)

    df = df[["countryiso3code", "country", "date", "value"]].copy()

    df["iso3"] = df["countryiso3code"]
    df["Country"] = df["country"].apply(lambda x: x.get("value") if isinstance(x, dict) else None)
    df["Year"] = df["date"]
    df[value_name] = pd.to_numeric(df["value"], errors="coerce")

    df = df[["iso3", "Country", "Year", value_name]].copy()

    df = df[df["Year"].isin(years)]
    df = df[df["Country"].isin(COUNTRY_ONLY)]
    df = df.dropna(subset=[value_name])

    return df

def clean_footnotes(x: str) -> str:
    if x is None:
        return x
    s = str(x)
    s = re.sub(r"\[[^\]]*\]", "", s) 
    return s.strip()

def coerce_numeric(series: pd.Series) -> pd.Series:
    s = series.astype(str).map(clean_footnotes)
    s = s.str.replace("–", "", regex=False)     
    s = s.str.replace(",", "", regex=False)
    s = s.str.strip()
    s = s.replace({"": np.nan, "-": np.nan, "—": np.nan})
    return pd.to_numeric(s, errors="coerce")

def flatten_columns(df: pd.DataFrame) -> pd.DataFrame:
    out = df.copy()
    if isinstance(out.columns, pd.MultiIndex):
        out.columns = [
            " ".join([str(x) for x in tup if str(x) != "nan"]).strip()
            for tup in out.columns
        ]
    out.columns = [str(c).strip() for c in out.columns]
    return out

def find_year_col(cols: list[str], year: int) -> str | None:
    y = str(year)
    for c in cols:
        if str(c).strip() == y:
            return c
    for c in cols:
        if y in str(c):
            return c
    return None

def add_iso3_from_name(df: pd.DataFrame, name_col: str, iso3_col: str = "iso3") -> pd.DataFrame:
    out = df.copy()

    iso3_overrides = {
        "Kosovo": "XKX",

        "São Tomé and Príncipe": "Sao Tome and Principe",
        "Turkey": "Turkiye",                       
        "Venezuela": "Venezuela, RB",
        "Vietnam": "Viet Nam",
        "Yemen": "Yemen, Rep.",                    
    }

    names = out[name_col].astype(str).map(clean_footnotes)

    def lookup_iso3(n: str):
        n = n.strip()

        if n in iso3_overrides:
            return iso3_overrides[n]

        if n in wb_name_to_iso3:
            return wb_name_to_iso3[n]

        n2 = name_to_wb.get(n, n)
        if n2 in wb_name_to_iso3:
            return wb_name_to_iso3[n2]

        n3 = " ".join(n2.split())
        return wb_name_to_iso3.get(n3, np.nan)

    out[iso3_col] = names.map(lookup_iso3)

    s = pd.Series(out[iso3_col], dtype="object")
    s = s.where(s.astype(str).str.fullmatch(r"[A-Z]{3}"), np.nan)
    out[iso3_col] = s

    return out

In contrast, the smooth term for alcohol consumption is not statistically significant, and the partial dependence plot shows only modest fluctuations around zero. The year factor is statistically significant, indicating that life expectancy increased over time even after adjusting for GDP and alcohol consumption. This reflects overall global improvements in health and living conditions between 1996 and 2019.
